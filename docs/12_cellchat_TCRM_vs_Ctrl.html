Run CellChat for Ctrl

# Subset Controls
seurat_LMC <- subset(seuratM, cond %in% "LMC")
sce_LMC <- as.SingleCellExperiment(seurat_LMC)
rownames(sce_LMC) = gsub("^.*\\.","",rownames(sce_LMC))

# Create CellChat object
cellchat_LMC <- createCellChat(object = sce_LMC, group.by = "clusterLabel")

# Select database
CellChatDB <- CellChatDB.mouse
CellChatDB.use <- CellChatDB
cellchat_LMC@DB <- CellChatDB.use

# Run analysis
cellchat_LMC <- subsetData(cellchat_LMC)
cellchat_LMC <- identifyOverExpressedGenes(cellchat_LMC)
cellchat_LMC <- identifyOverExpressedInteractions(cellchat_LMC)

# Filter
options(future.globals.maxSize = 8000 * 1024^2) #necessary if limit too small
cellchat_LMC <- filterCommunication(cellchat_LMC, min.cells = 5)
cellchat_LMC <- computeCommunProb(cellchat_LMC, type =  "truncatedMean", trim = 0.2, population.size = FALSE)

cellchat_LMC <- computeCommunProbPathway(cellchat_LMC)

cellchat_LMC <- aggregateNet(cellchat_LMC)

future.seed=TRUE
cellchat_LMC <- netAnalysis_computeCentrality(cellchat_LMC, slot.name = "netP")

Save CellChat Ctrl

basedir <- here()
saveRDS(cellchat_LMC, file = paste0(basedir, "/data/pericarditis_allmerged_cleaned_cellchat_LMC.rds"))
