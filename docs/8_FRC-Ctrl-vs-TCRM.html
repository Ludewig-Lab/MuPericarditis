# Set identity
Idents(seuratFB) <- seuratFB$FBclusterLabel

clusters <- levels(seuratFB)
conditions <- unique(seuratFB$cond)

DEgenes_cond_list <- list()

for (cl in clusters) {
  cat("Processing cluster:", cl, "\n")
  
  # Subset to this cluster
  sub <- subset(seuratFB, idents = cl)
  
  # Set condition as identity class
  Idents(sub) <- sub$cond
  
  # Ensure both conditions exist in this cluster
  cond_present <- intersect(conditions, levels(Idents(sub)))
  if (length(cond_present) < 2) {
    cat("   Skipping — not both conditions present in", cl, "\n")
    next
  }
  
  # Differential expression: TCRM vs LMC
  deg <- FindMarkers(
    sub,
    ident.1 = "TCRM",
    ident.2 = "LMC",
    only.pos = FALSE,
    logfc.threshold = 0,
    min.pct = 0.1
  )
  
  # Add gene column and extract EnsID
  deg <- deg %>%
    tibble::rownames_to_column(var = "gene") %>%
    mutate(
      EnsID = sub("\\..*", "", gene),  # clean Ensembl ID
      Cluster = cl
    )
  
  # Split into condition-specific sets
  deg_up_TCRM <- deg %>% filter(avg_log2FC > 0) %>% mutate(Condition = "TCRM")
  deg_up_LMC  <- deg %>% filter(avg_log2FC < 0) %>% mutate(Condition = "LMC")
  
  # Store
  DEgenes_cond_list[[paste0(cl, "_TCRM")]] <- deg_up_TCRM
  DEgenes_cond_list[[paste0(cl, "_LMC")]]  <- deg_up_LMC
}

# Combine enrichment results across clusters and conditions
plot_df <- do.call(
  rbind,
  lapply(names(ego_list), function(nm) {

    ego <- ego_list[[nm]]
    if (is.null(ego)) return(NULL)

    df <- ego@result

    # Extract cluster and condition from list name
    parts <- strsplit(nm, "\\.")[[1]]
    cluster <- parts[1]
    condition <- parts[2]

    df %>%
      filter(Description %in% sel_FRC) %>%
      mutate(
        Cluster   = cluster,
        Condition = condition,
        log10_padj = -log10(p.adjust)
      )
  })
)

# Filter for Ccl19 FRCs and conditions

  mutate(log10_padj = pmin(log10_padj, 10))

  group_by(Description) %>%
  mutate(max_log10_padj = max(log10_padj, na.rm = TRUE)) %>%

  arrange(desc(max_log10_padj)) %>%
