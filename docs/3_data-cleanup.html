# Function to create histograms per cluster
plotQC_perCluster <- function(sce, feature) {
  df <- as.data.frame(sce@colData)
  clusters <- unique(df$RNA_snn_res.0.25)  # get all clusters
  
  plots <- list()
  
  for (cl in clusters) {
    df_cl <- df[df$RNA_snn_res.0.25 == cl, ]
    
    p <- gghistogram(df_cl,
                     x = feature,
                     bins = 70,
                     fill = "#4682B4",
                     color = "#000000",
                     add_density = TRUE,
                     y = "..density..") +
      ggtitle(paste("Cluster", cl))
    
    plots[[as.character(cl)]] <- p
  }
  
  return(plots)
}
